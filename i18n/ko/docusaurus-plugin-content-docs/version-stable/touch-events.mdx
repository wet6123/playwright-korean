## Introduction

스와이프, 핀치, 탭과 같은 제스처에 응답하기 위해 레거시 [touch events](https://developer.mozilla.org/en-US/docs/Web/API/Touch_events)를 처리하는 웹 애플리케이션은 [TouchEvent](https://developer.mozilla.org/en-US/docs/Web/API/TouchEvent/TouchEvent)를 페이지에 수동으로 디스패치하여 테스트할 수 있습니다. 아래 예제는 [locator.dispatchEvent()](/api/class-locator.mdx#locator-dispatch-event)를 사용하여 [Touch](https://developer.mozilla.org/en-US/docs/Web/API/Touch) 포인트를 인수로 전달하는 방법을 보여줍니다.

### Emulating pan gesture

아래 예제에서는 지도를 이동할 것으로 예상되는 팬 제스처를 에뮬레이트합니다. 테스트 중인 앱은 터치 포인트의 `clientX/clientY` 좌표만 사용하므로 해당 좌표만 초기화합니다. 앱에 필요한 경우 더 복잡한 시나리오에서는 `pageX/pageY/screenX/screenY`도 설정해야 할 수 있습니다.

```js
import { test, expect, devices, type Locator } from '@playwright/test';

test.use({ ...devices['Pixel 7'] });

async function pan(locator: Locator, deltaX?: number, deltaY?: number, steps?: number) {
  const { centerX, centerY } = await locator.evaluate((target: HTMLElement) => {
    const bounds = target.getBoundingClientRect();
    const centerX = bounds.left + bounds.width / 2;
    const centerY = bounds.top + bounds.height / 2;
    return { centerX, centerY };
  });

  // 앱이 해당 좌표만 신경 쓰므로 clientX 및 clientY만 제공합니다.
  const touches = [{
    identifier: 0,
    clientX: centerX,
    clientY: centerY,
  }];
  await locator.dispatchEvent('touchstart',
      { touches, changedTouches: touches, targetTouches: touches });

  steps = steps ?? 5;
  deltaX = deltaX ?? 0;
  deltaY = deltaY ?? 0;
  for (let i = 1; i <= steps; i++) {
    const touches = [{
      identifier: 0,
      clientX: centerX + deltaX * i / steps,
      clientY: centerY + deltaY * i / steps,
    }];
    await locator.dispatchEvent('touchmove',
        { touches, changedTouches: touches, targetTouches: touches });
  }

  await locator.dispatchEvent('touchend');
}

test(`pan gesture to move the map`, async ({ page }) => {
  await page.goto('https://www.google.com/maps/place/@37.4117722,-122.0713234,15z',
      { waitUntil: 'commit' });
  await page.getByRole('button', { name: 'Keep using web' }).click();
  await expect(page.getByRole('button', { name: 'Keep using web' })).not.toBeVisible();
  // 지도 요소를 가져옵니다.
  const met = page.locator('[data-test-id="met"]');
  for (let i = 0; i < 5; i++)
    await pan(met, 200, 100);
  // 지도가 이동되었는지 확인합니다.
  await expect(met).toHaveScreenshot();
});
```

### Emulating pinch gesture

아래 예제에서는 핀치 제스처, 즉 두 개의 터치 포인트가 서로 더 가까워지는 것을 에뮬레이트합니다. 지도를 축소할 것으로 예상됩니다. 테스트 중인 앱은 터치 포인트의 `clientX/clientY` 좌표만 사용하므로 해당 좌표만 초기화합니다. 앱에 필요한 경우 더 복잡한 시나리오에서는 `pageX/pageY/screenX/screenY`도 설정해야 할 수 있습니다.

```js
import { test, expect, devices, type Locator } from '@playwright/test';

test.use({ ...devices['Pixel 7'] });

async function pinch(locator: Locator,
  arg: { deltaX?: number, deltaY?: number, steps?: number, direction?: 'in' | 'out' }) {
  const { centerX, centerY } = await locator.evaluate((target: HTMLElement) => {
    const bounds = target.getBoundingClientRect();
    const centerX = bounds.left + bounds.width / 2;
    const centerY = bounds.top + bounds.height / 2;
    return { centerX, centerY };
  });

  const deltaX = arg.deltaX ?? 50;
  const steps = arg.steps ?? 5;
  const stepDeltaX = deltaX / (steps + 1);

  // 요소의 중심에서 동일한 거리에 있는 두 개의 터치 포인트입니다.
  const touches = [
    {
      identifier: 0,
      clientX: centerX - (arg.direction === 'in' ? deltaX : stepDeltaX),
      clientY: centerY,
    },
    {
      identifier: 1,
      clientX: centerX + (arg.direction === 'in' ? deltaX : stepDeltaX),
      clientY: centerY,
    },
  ];
  await locator.dispatchEvent('touchstart',
      { touches, changedTouches: touches, targetTouches: touches });

  // 터치 포인트를 서로 가깝게 또는 멀어지게 이동합니다.
  for (let i = 1; i <= steps; i++) {
    const offset = (arg.direction === 'in' ? (deltaX - i * stepDeltaX) : (stepDeltaX * (i + 1)));
    const touches = [
      {
        identifier: 0,
        clientX: centerX - offset,
        clientY: centerY,
      },
      {
        identifier: 0,
        clientX: centerX + offset,
        clientY: centerY,
      },
    ];
    await locator.dispatchEvent('touchmove',
        { touches, changedTouches: touches, targetTouches: touches });
  }

  await locator.dispatchEvent('touchend', { touches: [], changedTouches: [], targetTouches: [] });
}

test(`pinch in gesture to zoom out the map`, async ({ page }) => {
  await page.goto('https://www.google.com/maps/place/@37.4117722,-122.0713234,15z',
      { waitUntil: 'commit' });
  await page.getByRole('button', { name: 'Keep using web' }).click();
  await expect(page.getByRole('button', { name: 'Keep using web' })).not.toBeVisible();
  // 지도 요소를 가져옵니다.
  const met = page.locator('[data-test-id="met"]');
  for (let i = 0; i < 5; i++)
    await pinch(met, { deltaX: 40, direction: 'in' });
  // 지도가 축소되었는지 확인합니다.
  await expect(met).toHaveScreenshot();
});
```
